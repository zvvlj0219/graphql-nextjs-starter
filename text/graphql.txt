①api routes 
pages/api/graphql.ts作成

// Queryのクエリ名①はスキーマの方のQueryのクエリ名②と一致していないとダメ
スキーマ
schema {
    query: Query
}

type Query {
    phones: [Phone]   ①ここでいうphones
}

type Phone {
    type: String
}

クエリ
query phone {  
    phones {    ②ここでいうphones
        type
    }
}
